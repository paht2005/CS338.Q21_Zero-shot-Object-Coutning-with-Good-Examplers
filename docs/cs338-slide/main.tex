\section{Tổng quan đề tài}

% --- 1.1 Mô tả bài toán ---
\begin{frame}[shrink=10]{1.1. Mô tả bài toán}
\scriptsize

\setbeamercolor{block title}{bg=uitblue!80,fg=white}
\setbeamercolor{block body}{bg=blue!5,fg=black}
\begin{block}{Định nghĩa}
\textbf{Object Counting} -- ước lượng số lượng đối tượng thuộc một lớp cho trước trong ảnh tự nhiên.\\[0.3em]
\textbf{Zero-shot Counting} -- chỉ cần ảnh + text prompt (tên lớp), \emph{không cần mẫu hình ảnh thủ công}. Hệ thống tự động sinh mẫu (exemplar) và tạo bản đồ mật độ (density map) để đếm.
\end{block}

\setbeamercolor{block title}{bg=green!70!black,fg=white}
\setbeamercolor{block body}{bg=green!5,fg=black}
\begin{block}{Ví dụ}
Ảnh chứa nhiều quả cam $+$ prompt \texttt{"oranges"} $\Rightarrow$ hệ thống trả về density map + tổng số cam.
\end{block}

\setbeamercolor{block title}{bg=orange!80!black,fg=white}
\setbeamercolor{block body}{bg=orange!10,fg=black}
\begin{block}{Ứng dụng}
\begin{itemize}
  \item Nông nghiệp thông minh (đếm quả, cây trồng).
  \item Giám sát giao thông, kiểm kê kho hàng.
  \item Phân tích hình ảnh y tế, phân tích đám đông.
\end{itemize}
\end{block}

\setbeamercolor{block title}{bg=red!80!black,fg=white}
\setbeamercolor{block body}{bg=red!10,fg=black}
\begin{block}{Thách thức}
\begin{itemize}
  \item Biến thiên lớn về mật độ và kích thước đối tượng.
  \item Đối tượng nhỏ, sát nhau $\Rightarrow$ khó tách ranh giới.
  \item Nền phức tạp, đối tượng tương tự $\Rightarrow$ nhầm lẫn tiền cảnh/nền.
\end{itemize}
\end{block}

\end{frame}

\begin{block}{Bối cảnh}
\begin{itemize}

  \item Các phương pháp hiện tại: phụ thuộc nhãn thủ công, hạn chế miền đối tượng, tính thủ công của few-shot.
\end{itemize}
\end{block}

\begin{block}{Mục tiêu đồ án}
\begin{itemize}
  \item \textbf{Tự động hóa} quy trình lấy mẫu: loại bỏ can thiệp thủ công.
  \item \textbf{Xây dựng} tương tác đa phương thức (text prompt $\leftrightarrow$ đặc trưng ảnh).
  \item \textbf{Tối ưu hóa} khả năng tổng quát hóa trên lớp đối tượng mới.
\end{itemize}
\end{block}

\scriptsize

\setbeamercolor{block title}{bg=uitblue!80,fg=white}

\begin{block}{Input}
\begin{itemize}

    \item Text prompt $T$ -- tên lớp tiếng Anh (ví dụ: \texttt{"oranges"}, \texttt{"keyboard keys"}).

\end{itemize}
\end{block}

\begin{frame}[fragile, shrink=15]{2.1. Kiến trúc tổng quan VA-Count (tiếp)}
\scriptsize

\begin{columns}[T,onlytextwidth]
\begin{column}{0.55\textwidth}

\setbeamercolor{block title}{bg=orange!80!black, fg=white}
\setbeamercolor{block body}{bg=orange!6!white}
\begin{block}{\textbf{3 bước của EEM}}
\textbf{Bước 1: Khởi tạo mẫu bằng GroundingDINO}
\begin{itemize}
  \item Nhánh dương: ảnh + tên lớp $\Rightarrow$ bbox ứng viên dương.
  \item Nhánh âm: ảnh + prompt chung $\Rightarrow$ tất cả vật thể.
\end{itemize}

\textbf{Bước 2: Tinh chỉnh \& lọc}
\begin{itemize}
  \item Loại trùng lặp (nhánh âm): IoU cao $\Rightarrow$ loại.
  \item Bộ lọc đơn đối tượng: CLIP ViT-B/32 classifier, $P > 0.8$.
\end{itemize}

\textbf{Bước 3: Chọn top-$k$ mẫu}
\begin{itemize}
  \item Mặc định: top-3 theo confidence score.
\end{itemize}
\end{block}
\end{column}

\begin{column}{0.42\textwidth}

\setbeamercolor{block title}{bg=violet!75!black, fg=white}
\setbeamercolor{block body}{bg=violet!5!white}
\begin{block}{\textbf{NSM \& Counter}}
\begin{enumerate}
  \item Feature ảnh = \textit{Query}; feature mẫu = \textit{Key/Value}.
  \item Attention + Decoder $\Rightarrow$ $D_p$, $D_n$.
  \item \textbf{Loss:}
\end{enumerate}
\vspace{-0.5em}
\[
\mathcal{L}_{C} = \mathrm{ReLU}(\mathcal{L}_p - \mathcal{L}_n + m)
\]
\[
\mathcal{L}_{\text{total}} = \mathcal{L}_{C} + \mathcal{L}_p
\]
{\tiny $m=0.5$, margin-ReLU contrastive.}
\end{block}

\end{column}
\end{columns}

\end{frame}

% --- 3.2 Độ đo đánh giá ---
\begin{frame}{3.2. Độ đo đánh giá}
\scriptsize

Hệ thống được đánh giá bằng hai độ đo (metric) chuẩn cho bài toán đếm:

\vspace{0.5em}

\begin{columns}[T,onlytextwidth]
\begin{column}{0.48\textwidth}

\setbeamercolor{block title}{bg=uitblue!80,fg=white}
\setbeamercolor{block body}{bg=blue!5,fg=black}
\begin{block}{\textbf{Mean Absolute Error (MAE)}}
Đo trung bình giá trị sai số tuyệt đối:
\[
\mathrm{MAE} = \frac{1}{N}\sum_{i=1}^{N}\left|y_i - \hat{y}_i\right|
\]
$N$: số ảnh, $y_i$: ground truth, $\hat{y}_i$: dự đoán.\\[0.3em]
$\Rightarrow$ Phản ánh \textbf{độ chính xác trung bình}.
\end{block}

\end{column}

\begin{column}{0.48\textwidth}

\setbeamercolor{block title}{bg=orange!80!black, fg=white}
\setbeamercolor{block body}{bg=orange!6!white}
\begin{block}{\textbf{Root Mean Squared Error (RMSE)}}
Phạt nặng các sai số lớn nhờ bình phương trước khi lấy căn:
\[
\mathrm{RMSE} = \sqrt{\frac{1}{N}\sum_{i=1}^{N}(y_i - \hat{y}_i)^2}
\]
$\Rightarrow$ Phản ánh \textbf{độ ổn định} trên các trường hợp khó (outlier).
\end{block}

\end{column}
\end{columns}

\vspace{0.5em}

\setbeamercolor{block title}{bg=green!70!black, fg=white}
\setbeamercolor{block body}{bg=green!5!white}
\begin{block}{\textbf{Nhận xét}}
Sử dụng đồng thời cả hai cho phép đánh giá toàn diện: MAE phản ánh độ chính xác trung bình; RMSE phản ánh độ ổn định trên các trường hợp khó.
\end{block}

\end{frame}

\section{Kết luận và hướng phát triển}

% --- 5.1 Kết luận ---
\begin{frame}[shrink=10]{5.1. Kết luận}
\scriptsize

\setbeamercolor{block title}{bg=blue!70!black, fg=white}
\setbeamercolor{block body}{bg=blue!5!white}
\begin{block}{\textbf{Về kiến trúc}}
VA-Count là thiết kế giàu thông tin học được, tách rõ giữa pha lấy mẫu (EEM) và pha sinh bản đồ mật độ (NSM). Nhánh dương/âm tạo điều kiện cho cơ chế học tương phản (contrastive), hữu hiệu cho zero-shot counting.
\end{block}

\setbeamercolor{block title}{bg=teal!70!black, fg=white}
\setbeamercolor{block body}{bg=teal!5!white}
\begin{block}{\textbf{Về cải tiến}}
\begin{itemize}\setlength{\itemsep}{2pt}
  \item \textbf{Rich Prompt:} hiệu quả lớn với YOLO-World ($19{,}03 \to 17{,}91$, $\Delta=1{,}12$), khiêm tốn với GDino ($\Delta=0{,}19$) do bão hòa ngữ nghĩa.
  \item \textbf{YOLO-World:} loại bỏ bottleneck tốc độ (trích xuất: 25h $\to$ 4h; demo: 0,60s/ảnh).
  \item \textbf{YOLO-World + Rich Prompt:} cân bằng tốt nhất -- MAE 17,91 với 2,4s/ảnh.
  \item \textbf{Phát hiện quan trọng:} bộ phát hiện càng yếu về ngữ nghĩa, Rich Prompt càng mang lại lợi ích lớn.
\end{itemize}
\end{block}

\setbeamercolor{block title}{bg=orange!80!black, fg=white}
\setbeamercolor{block body}{bg=orange!6!white}
\begin{block}{\textbf{Về cài đặt}}
\begin{itemize}\setlength{\itemsep}{1pt}
  \item Hàm tương phản dạng \emph{margin-ReLU} (tương đương softmax, ổn định số học hơn).
  \item CLIP ViT-L/14 cho re-ranking; CLIP ViT-B/32 cho bộ phân loại nhị phân.
  \item Gemini 2.0 Flash (\texttt{gemini-2.0-flash-exp}) thay Gemini 2.5 Flash.
\end{itemize}
\end{block}

\end{frame}

% --- 5.2 Hướng phát triển ---
\begin{frame}[shrink=10]{5.2. Hướng phát triển}
\scriptsize

\begin{columns}[T,onlytextwidth]

\begin{column}{0.48\textwidth}
\setbeamercolor{block title}{bg=teal!75!black, fg=white}
\setbeamercolor{block body}{bg=teal!5!white}
\begin{block}{\textbf{Ngắn hạn}}
\begin{itemize}\setlength{\itemsep}{2pt}
  \item \textbf{Khử phân mảnh:} graph-based clustering, watershed hoặc head phụ dự đoán instance-aware density.
  \item \textbf{Mật độ dày đặc:} multi-resolution / tile-based inference để giữ chi tiết tại biên vật thể sát nhau.
  \item \textbf{Khảo sát cài đặt:} softmax contrastive vs margin-ReLU; CLIP ViT-B/32 vs L/14; Gemini 2.5 Flash khi có quota.
\end{itemize}
\end{block}
\end{column}

\begin{column}{0.48\textwidth}
\setbeamercolor{block title}{bg=purple!70!black, fg=white}
\setbeamercolor{block body}{bg=purple!5!white}
\begin{block}{\textbf{Dài hạn}}
\begin{itemize}\setlength{\itemsep}{2pt}
  \item \textbf{Khử thiên lệch:} đối sánh trực tiếp (CounTR) thay density map MAE.
  \item \textbf{Regularizer:} ép output = 0 khi mẫu không hợp lệ.
  \item \textbf{Triển khai:} Docker + HuggingFace Spaces / Vercel deployment.
\end{itemize}
\end{block}
\end{column}

\end{columns}

\end{frame}
